GCS 하이브리드 스토리지 시스템 run/core/config.py

문제: Cloud Run의 일시적 스토리지 한계로 상태 유지 불가

해결: GCS를 주 저장소로, 로컬을 백업으로 사용하는 하이브리드 전략. 5초 캐시로 성능 최적화

# 설정 캐시 (성능 개선)
settings_cache = None
cache_timestamp = 0
CACHE_DURATION = 5  # 5초 캐시

def load_settings():
    """GCS 또는 로컬 백업에서 설정 파일을 로드합니다. (GCS 우선)"""
    global settings_cache, cache_timestamp

    current_time = time.time()

    # 캐시된 설정이 있고 아직 유효하면 사용
    if settings_cache is not None and (current_time - cache_timestamp) < CACHE_DURATION:
        return settings_cache.copy()

    # 1순위: GCS에서 로드
    client = get_gcs_client()
    if client:
        try:
            bucket = client.bucket(GCS_BUCKET)
            blob = bucket.blob(GCS_KEY)
            settings = json.loads(blob.download_as_text())
            settings_cache = settings.copy()
            cache_timestamp = current_time
            return settings
        except Exception as e:
            print(f"[경고] GCS 로드 실패: {e}")

    # 2순위: 로컬 백업 파일에서 로드 (GCS 실패 시 폴백)
    if os.path.exists(backup_file):
        with open(backup_file, 'r', encoding='utf-8') as f:
            settings = json.load(f)
        return settings

    # 모두 실패 시 기본 구조 반환
    return {"guilds": {}, "users": {}, "global": {}}

Qwen3-TTS Modal 서버리스 GPU 배포 tts_modal_deploy.py

기능: 화자별 파인튜닝된 모델로 캐릭터 음성 합성

AI 요소: Qwen3-TTS + Transformers 기반 화자 적응 (Speaker Adaptation)

인프라: Modal 서버리스 GPU로 비용 효율적 배포 (사용한 만큼만 과금)

import modal

# Modal 앱 정의
app = modal.App("qwen3-tts")

# GPU 이미지 설정
image = modal.Image.debian_slim(python_version="3.11").pip_install(
    "torch", "transformers", "soundfile", "qwen-tts"
)

# 화자별 파인튜닝 모델
SPEAKER_MODELS = {
    "debi": "checkpoint-epoch-9",      # 데비 모델
    "marlene": "checkpoint-epoch-27"   # 마를렌 모델
}

@app.cls(gpu="T4", image=image)
class TTSService:
    @modal.enter()
    def load_models(self):
        """컨테이너 시작 시 모델 로드"""
        from qwen_tts import Qwen3TTSModel
        self.models = {}
        for speaker, path in SPEAKER_MODELS.items():
            self.models[speaker] = Qwen3TTSModel.from_pretrained(path)

    @modal.method()
    def generate(self, text: str, speaker: str = "debi") -> bytes:
        """텍스트를 음성으로 변환"""
        model = self.models[speaker]
        wavs, sr = model.generate_custom_voice(text=text, speaker=speaker)

        # WAV 바이트로 변환
        import io, soundfile as sf
        buffer = io.BytesIO()
        sf.write(buffer, wavs[0], sr, format="WAV")
        return buffer.getvalue()
